294  std::string Option_<T>::match(const std::string& args) const {
295  std::string parsable(args);
296  std::string shortNameString(std::string("") + shortName);
297  std::size_t startMatch;
298  const auto longSequence = std::string(" --") + longName + " ";
299  const auto shortSequence = std::string(" -") + shortName + " ";
300 
301  unless ((startMatch = parsable.find(longSequence)) == -1) {
302  handleParameterParsing(startMatch, parsable, longName, longSequence);
303  } else unless ((startMatch = parsable.find(shortSequence)) == -1) {
304  handleParameterParsing(startMatch, parsable, shortNameString, shortSequence);
305  }
306 
307  return parsable;
308  }

342  std::string Option_<T>::_getOptionValueAsString(std::size_t startMatch,
343  std::string& args,
344  const std::string& name,
345  const std::string& sequence) const {
346  std::size_t matchEnd = args.find(' ', startMatch + sequence.length() + 1);
347  std::size_t matchLength = matchEnd - startMatch;
348  std::string match = args.substr(startMatch, matchLength);
349  args.erase(startMatch, matchLength);
